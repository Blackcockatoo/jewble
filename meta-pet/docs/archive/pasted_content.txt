    const state = {
      form: 'explorer',

    };

    const paletteDefs = {
      iris: {
        base: '#1d2345',
        glow: '#6366f1',
        accent: '#a5b4fc',
        secondary: '#22c1c3',
      },
      ember: {

    const forms = ['explorer', 'sleep', 'study', 'battle'];
    const moods = ['calm', 'focused', 'agitated'];
    const palettes = Object.keys(paletteDefs);

    function lerp(a, b, t) {
      return a + (b - a) * t;
    }

    function clamp01(x) {
      return Math.max(0, Math.min(1, x));
    }

    function render() {
      const svg = document.getElementById('petSvg');
      const { form, mood, palette, energy, curiosity } = state;
      const colors = paletteDefs[palette];

      const energyN = clamp01(energy / 100);
      const curiosityN = clamp01(curiosity / 100);

      const isAsleep = form === 'sleep';
      const isBattle = form === 'battle';
      const isStudy = form === 'study';

      const baseRy = isAsleep ? 60 : isBattle ? 50 : 60;
      const baseRx = isAsleep ? 50 : isBattle ? 45 : 55;
      const cy = isAsleep ? 160 : 150;

      // Body squish based on energy: low energy => more squished & droopy
      const squish = lerp(0.0, 0.18, 1 - energyN);
      const rx = baseRx * (1 + squish * 0.4);
      const ry = baseRy * (1 - squish * 0.4);

      const glowOpacity = lerp(0.35, 1.0, energyN);
      const pulseDur = lerp(5.5, 2.2, energyN).toFixed(1);

      const orbitSpeed = lerp(20, 6, curiosityN).toFixed(1);

      const moodTilt = mood === 'calm' ? 0 : mood === 'focused' ? -5 : 6;

      const eyeOpen = isAsleep
        ? 0
        : lerp(0.35, 1, curiosityN * (mood === 'agitated' ? 1.05 : 0.9));

      const bgGradTop = colors.glow;
      const bgGradMid = colors.base;
      const bgGradBottom = '#020617';

      const orbitCount = form === 'explorer' ? 4 : form === 'battle' ? 3 : 2;

      const sleepMotes = form === 'sleep' ? 5 : 0;

          </radialGradient>
